'use strict';

var express = require('express');
var controller = require('./main.controller');

var router = express.Router();

/**
* Main module
* @module api/main
*/ 

/**
* Route for the home page
*
*/ /*
* @function router.get
* @param {string} / path to home page
* @param {function} controller.index
*/
router.get('/', controller.index);


/**
*Join a party, get routes
*
*/ /*
* @function router.get
* @param {string} /join path to join a party page
* @param {function} controller.join_get
*/

router.get('/join', controller.join_get);

/**
*Join a party, post route
*
*/ /*
* @function router.post
* @param {string} /join path to join a party page
* @param {function} controller.join_post
*/

router.post('/join', controller.join_post);

/**
*create a party, get route
*
*/ /*
* @function router.post
* @param {string} /host 
* @param {function} controller.host_get
*/

router.get('/host', controller.host_get);

/**
*Create a party, post route
*
*/ /*
* @function router.post
* @param {string} /host 
* @param {function} controller.host_post
*/
router.post('/host', controller.host_post);

/**
* The party view, angular app
*
*/ /*
* @function router.get
* @param {string} /p/:name name of party
* @param {function} controller.party
*
*/

router.get('/p/:name', controller.party);

module.exports = router;
